* The First Driver
** A Bare Minimum Driver
We are writing a bare minimum RTL8139 driver. It does nothing but announces its existence as a Loadable Kernel Module (LKM). The purpose of writing it is testing whether we can build and run an LKM on the VM.

/File:/ =src/ch03/rtl8139_driver.c=
#+begin_src c
  #include <linux/init.h>
  #include <linux/kernel.h>
  #include <linux/module.h>

  MODULE_LICENSE("GPL");
  MODULE_AUTHOR("Zengxu Yang");
  MODULE_DESCRIPTION("A simple RTL8139 PCI Ethernet driver from scratch.");

  static int __init rtl8139_init(void)
  {
  	printk(KERN_INFO "rtl8139: Driver module loaded successfully.\n");
  	return 0;
  }

  static void __exit rtl8139_exit(void)
  {
  	printk(KERN_INFO "rtl8139: Driver module unloaded.\n");
  }

  module_init(rtl8139_init);
  module_exit(rtl8139_exit);
#+end_src
/File:/ =src/ch03/Makefile=
#+begin_src makefile
  obj-m += rtl8139_driver.o

  KDIR := /lib/modules/$(shell uname -r)/build
  PWD  := $(shell pwd)

  default:
  	$(MAKE) -C $(KDIR) M=$(PWD) modules

  clean:
  	$(MAKE) -C $(KDIR) M=$(PWD) clean
#+end_src
